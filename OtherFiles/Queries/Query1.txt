db.logainm.aggregate([
	{
		$unwind: "$glossary"
	},
	{
		$group: {
			_id: {
                		headword: "$glossary.headword",
                		translation: "$glossary.translation"
            		},
			count: {
				$sum: 1
			}
		}
	},
	{
		$sort: {
			count: -1
		}
	},
	{
		$limit: 10
	}
])
